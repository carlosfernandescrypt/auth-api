<script>
  // Função de login
  function realizar_login(event) {
    event.preventDefault(); // Previne o comportamento padrão do formulário

    const username = document.getElementById("username").value;
    const password = document.getElementById("password").value;

    // Fazendo a requisição de login
    fetch("https://8aab-179-124-14-78.ngrok-free.app/api/auth/login", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ username, password }),
    })
      .then((response) => {
        if (!response.ok) {
          throw new Error("Falha no login");
        }
        return response.json();
      })
      .then((data) => {
        localStorage.setItem("token", data.token);
        console.log("Login bem-sucedido!");

        // Redireciona o cliente para a página específica
        window.location.href =
          "https://areacliente.oneelevadores.com.br/area-do-cliente";
      })
      .catch((error) => {
        console.error("Erro ao realizar login:", error);
        alert("Usuário ou senha inválidos.");
      });
  }

  // Adiciona um event listener para o formulário
  document
    .getElementById("login-form")
    .addEventListener("submit", realizar_login);
</script>
